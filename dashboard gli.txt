def load_state() -> dict:
    """Load current bot state from shared JSON file."""
    try:
        if os.path.exists(STATE_FILE):
            with open(STATE_FILE) as f:
                return json.load(f)
    except Exception:
        pass
    # Default demo state when bot isn't running
    return {
        "running": False,
        "paused": False,
        "mode": "paper",
        "capital": 1000.0,
        "start_capital": 1000.0,
        "btc_price": 0.0,
        "btc_change_1m": 0.0,
        "btc_change_5m": 0.0,
        "wins": 0,
        "losses": 0,
        "open_trades": [],
        "markets_tracked": 0,
        "last_latency_ms": 0,
        "scan_count": 0,
        "consecutive_losses": 0,
        "claude_enabled": False,
        "claude_regime": "unknown",
        "claude_multiplier": 1.0,
        "uptime_seconds": 0,
        "pnl_history": [],
        "last_updated": None
    }

def load_trades() -> list:
    """Load closed trade history."""
    try:
        if os.path.exists(TRADES_FILE):
            with open(TRADES_FILE) as f:
                data = json.load(f)
                return data.get("closed", [])
    except Exception:
        pass
    return []

def load_logs(n=40) -> list:
    """Load last N lines from bot log."""
    try:
        if os.path.exists(LOG_FILE):
            with open(LOG_FILE) as f:
                lines = f.readlines()
                return lines[-n:]
    except Exception:
        pass
    return ["[no log file found — start the bot first]"]

def send_command(cmd: dict):
    """Write a command for the bot to pick up."""
    try:
        with open(COMMAND_FILE, "w") as f:
            json.dump({**cmd, "timestamp": time.time()}, f)
    except Exception as e:
        st.error(f"Could not write command: {e}")

def fmt_pct(v: float, decimals=3) -> str:
    sign = "+" if v >= 0 else ""
    return f"{sign}{v:.{decimals}f}%"

def fmt_price(v: float) -> str:
    return f"${v:,.2f}"

def fmt_uptime(seconds: int) -> str:
    h = seconds // 3600
    m = (seconds % 3600) // 60
    s = seconds % 60
    return f"{h:02d}:{m:02d}:{s:02d}"

@st.cache_data(ttl=60)
def fetch_btc_candles(limit: int = 30) -> pd.DataFrame:
    """Fetch BTC/USDT 1-minute candles from Binance REST (public, no auth)."""
    try:
        url = "https://api.binance.com/api/v3/klines"
        resp = requests.get(url, params={"symbol": "BTCUSDT", "interval": "1m", "limit": limit}, timeout=5)
        resp.raise_for_status()
        raw = resp.json()
        df = pd.DataFrame(raw, columns=[
            "open_time", "open", "high", "low", "close", "volume",
            "close_time", "quote_vol", "trades", "taker_buy_base",
            "taker_buy_quote", "ignore"
        ])
        for col in ["open", "high", "low", "close", "volume"]:
            df[col] = df[col].astype(float)
        df["time"] = pd.to_datetime(df["open_time"], unit="ms")
        return df[["time", "open", "high", "low", "close", "volume"]]
    except Exception:
        return pd.DataFrame()

def render_header(state: dict):
    pnl     = state["capital"] - state["start_capital"]
    pnl_pct = pnl / state["start_capital"] * 100 if state["start_capital"] else 0
    total   = state["wins"] + state["losses"]
    wr      = state["wins"] / total * 100 if total > 0 else 0

    st.markdown("## PolyBot · BTC 5-Min Trader")

    col1, col2, col3, col4, col5, col6 = st.columns(6)

    with col1:
        st.metric(
            "Capital",
            fmt_price(state["capital"]),
            delta=f"{fmt_pct(pnl_pct)} (${pnl:+.2f})",
            delta_color="normal"
        )
    with col2:
        st.metric(
            "BTC Price",
            fmt_price(state["btc_price"]) if state["btc_price"] > 0 else "—",
            delta=fmt_pct(state["btc_change_1m"] * 100) + " 1m"
        )
    with col3:
        st.metric(
            "Win Rate",
            f"{wr:.1f}%" if total > 0 else "—",
            delta=f"{total} trades"
        )
    with col4:
        st.metric(
            "Open Positions",
            len(state.get("open_trades", [])),
            delta="max 3"
        )
    with col5:
        lat = state.get("last_latency_ms", 0)
        st.metric(
            "Latency",
            f"{lat:.0f}ms" if lat > 0 else "—",
            delta="target <100ms",
            delta_color="inverse" if lat > 100 else "normal"
        )
    with col6:
        st.metric(
            "Markets",
            state.get("markets_tracked", 0),
            delta="active BTC contracts"
        )

def render_open_trades(state: dict):
    st.markdown('<div class="section-label">Open Positions</div>', unsafe_allow_html=True)
    trades = state.get("open_trades", [])

    if not trades:
        st.caption("No open positions right now.")
        return

    for t in trades:
        side    = t.get("side", "?")
        q       = t.get("question", "Unknown market")[:45]
        entry   = t.get("entry_price", 0)
        shares  = t.get("shares", 0)
        cur     = t.get("current_price", entry)
        pnl     = (cur - entry) * shares
        held    = int(time.time() - t.get("entry_time", time.time()))
        pnl_cls = "pnl-pos" if pnl >= 0 else "pnl-neg"
        side_color = "#48bb78" if side == "YES" else "#fc8181"

        st.markdown(
            f"""<div class="trade-row">
              <span style="color:{side_color};font-size:10px;font-weight:700;min-width:28px">{side}</span>
              <span style="flex:1;color:#8899aa;font-size:10px;margin:0 8px">{q}</span>
              <span style="color:#4a5568;font-size:10px;min-width:30px">{held}s</span>
              <span class="{pnl_cls}" style="font-size:11px;min-width:60px;text-align:right">{'+' if pnl>=0 else ''}{pnl:.4f}</span>
            </div>""",
            unsafe_allow_html=True
        )


def render_trade_history(trades: list):
    st.markdown('<div class="section-label">Closed Trades</div>', unsafe_allow_html=True)

    if not trades:
        st.caption("No closed trades yet. They'll appear here during paper trading.")
        return

    df = pd.DataFrame(trades[-50:])  # Show last 50
    if df.empty:
        return

    # Format for display
    display_cols = {
        "trade_id": "ID",
        "side": "Side",
        "entry_price": "Entry",
        "exit_price": "Exit",
        "pnl": "P&L",
        "pnl_pct": "P&L %",
        "exit_reason": "Reason"
    }
    available = [c for c in display_cols if c in df.columns]
    df_show = df[available].copy().rename(columns=display_cols)

    if "Entry" in df_show.columns:
        df_show["Entry"] = df_show["Entry"].apply(lambda x: f"${x:.4f}")
    if "Exit" in df_show.columns:
        df_show["Exit"] = df_show["Exit"].apply(lambda x: f"${x:.4f}")
    if "P&L" in df_show.columns:
        df_show["P&L"] = df_show["P&L"].apply(lambda x: f"{'+'if x>=0 else ''}{x:.4f}")
    if "P&L %" in df_show.columns:
        df_show["P&L %"] = df_show["P&L %"].apply(lambda x: f"{'+'if x>=0 else ''}{x*100:.3f}%")

    # Reverse to show newest first
    df_show = df_show.iloc[::-1]

    st.dataframe(
        df_show,
        use_container_width=True,
        height=220,
        hide_index=True
    )

    # CSV export
    csv = pd.DataFrame(trades).to_csv(index=False)
    st.download_button(
        "⬇ Export all trades (.csv)",
        data=csv,
        file_name=f"trades_{datetime.now().strftime('%Y%m%d_%H%M')}.csv",
        mime="text/csv",
        use_container_width=True,
    )


def render_win_rate_chart(trades: list):
    st.markdown('<div class="section-label">Win / Loss Breakdown</div>', unsafe_allow_html=True)

    if not trades:
        st.caption("No data yet.")
        return

    df = pd.DataFrame(trades)
    wins   = len(df[df["pnl"] > 0]) if "pnl" in df.columns else 0
    losses = len(df[df["pnl"] <= 0]) if "pnl" in df.columns else 0

    fig = go.Figure(data=[go.Pie(
        labels=["Wins", "Losses"],
        values=[wins, losses],
        hole=0.65,
        marker_colors=["#48bb78", "#fc8181"],
        textfont_size=11,
        textfont_color="#c8d0dc"
    )])
    fig.update_layout(
        paper_bgcolor="rgba(0,0,0,0)",
        plot_bgcolor="rgba(0,0,0,0)",
        height=180,
        margin=dict(l=0, r=0, t=0, b=0),
        showlegend=True,
        legend=dict(
            font=dict(color="#8899aa", size=10),
            bgcolor="rgba(0,0,0,0)"
        ),
        annotations=[dict(
            text=f"{wins/(wins+losses)*100:.0f}%<br>win" if (wins+losses) > 0 else "—",
            x=0.5, y=0.5, showarrow=False,
            font=dict(size=14, color="#e2e8f0")
        )]
    )
    st.plotly_chart(fig, use_container_width=True)

def render_exit_reason_chart(trades: list):
    st.markdown('<div class="section-label">Exit Reasons</div>', unsafe_allow_html=True)

    if not trades:
        st.caption("No data yet.")
        return

    df = pd.DataFrame(trades)
    if "exit_reason" not in df.columns or "pnl" not in df.columns:
        st.caption("No data yet.")
        return

    reason_colors = {
        "profit_target":    "#48bb78",
        "stop_loss":        "#fc8181",
        "max_hold_timeout": "#ed8936",
        "expiry_approaching": "#a78bfa",
        "signal_reversal":  "#4a9eff",
        "market_not_found": "#4a5568",
    }

    grouped = df.groupby("exit_reason").agg(
        count=("pnl", "count"),
        total_pnl=("pnl", "sum")
    ).reset_index()

    fig = go.Figure()
    for _, row in grouped.iterrows():
        reason = row["exit_reason"] or "unknown"
        color  = reason_colors.get(reason, "#4a5568")
        fig.add_trace(go.Bar(
        streaame=reason,
            x=[reason.replace("_", " ")],
            y=[row["count"]],
            marker_color=color,
            text=[f"${row['total_pnl']:+.3f}"],
            textposition="outside",
            textfont=dict(size=9, color="#8899aa"),
        ))
    fig.update_layout(
        paper_bgcolor="rgba(0,0,0,0)",
        plot_bgcolor="rgba(0,0,0,0)",
        height=180,
        margin=dict(l=0, r=0, t=8, b=0),
        showlegend=False,
        barmode="group",
        xaxis=dict(showgrid=False, color="#4a5568", tickfont=dict(size=9)),
        yaxis=dict(showgrid=True, gridcolor="#1e2530", color="#4a5568", tickfont=dict(size=9)),
    )
    st.plotly_chart(fig, use_container_width=True)


def render_duration_histogram(trades: list):
    st.markdown('<div class="section-label">Hold Duration</div>', unsafe_allow_html=True)

    if not trades:
        st.caption("No data yet.")
        return

    df = pd.DataFrame(trades)
    if "entry_time" not in df.columns or "exit_time" not in df.columns:
        st.caption("No data yet.")
        return

    df = df[(df["exit_time"] > 0) & (df["entry_time"] > 0)].copy()
    df["duration_s"] = (df["exit_time"] - df["entry_time"]).clip(lower=0)

    if df.empty:
        st.caption("No closed trades yet.")
        return

    fig = go.Figure()
    fig.add_trace(go.Histogram(
        x=df["duration_s"],
        nbinsx=20,
        marker_color="#4a9eff",
        marker_line_color="#1e2530",
        marker_line_width=1,
        opacity=0.8,
    ))
    fig.update_layout(
        paper_bgcolor="rgba(0,0,0,0)",
        plot_bgcolor="rgba(0,0,0,0)",
        height=180,
        margin=dict(l=0, r=0, t=8, b=0),
        xaxis=dict(showgrid=False, color="#4a5568", tickfont=dict(size=9), title_text="seconds", title_font=dict(size=9, color="#4a5568")),
        yaxis=dict(showgrid=True, gridcolor="#1e2530", color="#4a5568", tickfont=dict(size=9)),
        bargap=0.05,
    )
    avg = df["duration_s"].mean()
    fig.add_vline(x=avg, line_dash="dash", line_color="#ed8936", line_width=1,
                  annotation_text=f"avg {avg:.0f}s", annotation_font=dict(size=9, color="#ed8936"),
                  annotation_position="top right")
    st.plotly_chart(fig, use_container_width=True)

def render_risk_meters(state: dict):
    st.markdown('<div class="section-label">Risk Meters</div>', unsafe_allow_html=True)

    capital       = state["capital"]
    start_capital = state["start_capital"]
    pnl_pct       = (capital - start_capital) / start_capital * 100 if start_capital else 0
    daily_loss_used = min(abs(min(pnl_pct, 0)) / 2.0 * 100, 100)
    consec        = state.get("consecutive_losses", 0)
    consec_pct    = consec / 5 * 100
    open_pos      = len(state.get("open_trades", []))
    cap_used_pct  = open_pos / 3 * 100

    def risk_bar(label, pct, fmt_val, danger_threshold=70):
        color = "#48bb78" if pct < 40 else ("#ed8936" if pct < danger_threshold else "#fc8181")
        st.markdown(f"""
        <div style="display:flex;align-items:center;gap:8px;margin-bottom:8px;font-size:11px">
          <span style="width:110px;color:#4a5568">{label}</span>
          <div style="flex:1;height:3px;background:#1e2530;border-radius:2px">
            <div style="width:{pct:.0f}%;height:3px;background:{color};border-radius:2px;transition:width 0.4s"></div>
          </div>
          <span style="width:40px;text-align:right;color:{color}">{fmt_val}</span>
        </div>""", unsafe_allow_html=True)

    risk_bar("Daily loss", daily_loss_used, f"{daily_loss_used:.0f}%", 70)
    risk_bar("Consec. losses", consec_pct, f"{consec}/5", 60)
    risk_bar("Positions open", cap_used_pct, f"{open_pos}/3", 80)

    if consec >= 5:
        st.error("⚡ Circuit breaker active — bot paused 10min")
    elif pnl_pct <= -2.0:
        st.error("🛑 Daily loss limit hit (−2%)")


def render_log(state: dict):
    st.markdown('<div class="section-label">Signal Log</div>', unsafe_allow_html=True)

    lines = load_logs(50)

    # Color-code log lines
    def colorize(line: str) -> str:
        line = line.strip()
        if "ERROR" in line:
            return f'<span style="color:#fc8181">{line}</span>'
        elif "ENTER" in line or "✅" in line:
            return f'<span style="color:#48bb78">{line}</span>'
        elif "EXIT" in line:
            return f'<span style="color:#4a9eff">{line}</span>'
        elif "RISK" in line or "WARNING" in line or "circuit" in line.lower():
            return f'<span style="color:#ed8936">{line}</span>'
        elif "CLAUDE" in line or "ClaudeBot" in line:
            return f'<span style="color:#a78bfa">{line}</span>'
        elif "SUCCESS" in line or "connected" in line.lower():
            return f'<span style="color:#68d391">{line}</span>'
        else:
            return f'<span style="color:#4a5568">{line}</span>'

    html_lines = "<br>".join(colorize(l) for l in lines[-40:])
    st.markdown(
        f'<div class="log-container">{html_lines}</div>',
        unsafe_allow_html=True
    )


def render_stats_summary(trades: list, state: dict):
    st.markdown('<div class="section-label">Session Stats</div>', unsafe_allow_html=True)

    if not trades:
        st.caption("Stats will appear after first closed trade.")
        return

    df = pd.DataFrame(trades)
    if "pnl" not in df.columns:
        return

    wins_df   = df[df["pnl"] > 0]["pnl"]
    losses_df = df[df["pnl"] <= 0]["pnl"]

    col1, col2 = st.columns(2)
    with col1:
        st.markdown(f"**Total P&L:** `${df['pnl'].sum():+.4f}`")
        st.markdown(f"**Avg win:** `${wins_df.mean():+.4f}`" if len(wins_df) else "**Avg win:** `—`")
        st.markdown(f"**Best trade:** `${df['pnl'].max():+.4f}`")
    with col2:
        st.markdown(f"**Total trades:** `{len(df)}`")
        st.markdown(f"**Avg loss:** `${losses_df.mean():+.4f}`" if len(losses_df) else "**Avg loss:** `—`")
        st.markdown(f"**Worst trade:** `${df['pnl'].min():+.4f}`")

    # Profit factor
    if len(losses_df) and losses_df.sum() != 0:
        pf = abs(wins_df.sum() / losses_df.sum())
        pf_color = "green" if pf >= 1.5 else ("orange" if pf >= 1.0 else "red")
        st.markdown(f"**Profit factor:** :{pf_color}[{pf:.2f}]")

    # Ready to go live?
    wr = len(wins_df) / len(df) * 100 if len(df) > 0 else 0
    st.markdown("---")
    st.markdown("**Go-live readiness:**")

    checks = [
        ("Win rate > 52%",     wr >= 52,        f"{wr:.1f}%"),
        ("Total trades > 50",  len(df) >= 50,   str(len(df))),
        ("No daily loss hit",  state.get("consecutive_losses", 0) < 5, "✓" if state.get("consecutive_losses",0) < 5 else "✗"),
        ("Profit factor > 1.5", len(losses_df) and abs(wins_df.sum()/losses_df.sum()) >= 1.5 if losses_df.sum()!=0 else False, "—")
    ]
    for label, passed, val in checks:
        icon = "✅" if passed else "⬜"
        st.markdown(f"{icon} {label} — `{val}`")
